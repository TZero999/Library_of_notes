"?" == {0,1} = 0 | 1    = "Ноль или один";
"*" == {0,}  = >=0      = "Ноль и больше".
"+" == {1,}  = >=1      = "Один и больше";

+, *, ? - "Жадные". Захватывают всю область поиска, постепенно уменьшают её:
str = '"Witch" and "Broom"', regexp = /".+"/g, result = [ "Witch" and "Broom" ]
+?, *?, ?? - "Ленивые". Поиск производится посимвольно, слева направо:
str = '"Witch" and "Broom"', regexp = /".+?"/g, result = [ "Witch" , "Broom" ]
